ğŸ“‹ Complete Example Program

Example: Here's a complete example fetching data: 
// UserList.js
import { useState, useEffect } from 'react';
import './UserList.css';

function UserList() {
  const [users, setUsers] = useState([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);
  
  useEffect(() => {
    // Using a free API for demonstration
    fetch('https://jsonplaceholder.typicode.com/users')
      .then(response => {
        if (!response.ok) {
          throw new Error('Failed to fetch users');
        }
        return response.json();
      })
      .then(data => {
        setUsers(data);
        setLoading(false);
      })
      .catch(err => {
        setError(err.message);
        setLoading(false);
      });
  }, []);
  
  if (loading) {
    return <div className="loading">Loading users...</div>;
  }
  
  if (error) {
    return <div className="error">Error: {error}</div>;
  }
  
  return (
    <div className="user-list-container">
      <h2>User List</h2>
      <div className="users-grid">
        {users.map(user => (
          <div key={user.id} className="user-card">
            <h3>{user.name}</h3>
            <p><strong>Email:</strong> {user.email}</p>
            <p><strong>Username:</strong> {user.username}</p>
            <p><strong>City:</strong> {user.address.city}</p>
          </div>
        ))}
      </div>
    </div>
  );
}

export default UserList;

ğŸ’¡ Key Concepts

  • fetch(): JavaScript function to get data from APIs
  • Promise: Represents a value that will be available later
  • then(): Handles successful responses
  • catch(): Handles errors
  • Loading state: Shows while data is being fetched
  • Error state: Shows if something went wrong
  • async/await: Alternative way to handle promises
