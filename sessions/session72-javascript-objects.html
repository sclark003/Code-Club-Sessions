üöÄ Activity 1: Creating Objects

Step 1: Create objects with properties: 
let person = {
    name: "Sarah",
    age: 12,
    school: "Code Club School",
    isStudent: true
};
Step 2: Access properties: 
console.log(person.name);  // "Sarah"
console.log(person['age']);  // 12

üíª Activity 2: Object Methods

Step 1: Add functions to objects: 
let person = {
    name: "Sarah",
    age: 12,
    greet: function() {
        return "Hello, I'm " + this.name;
    }
};

console.log(person.greet());

<!DOCTYPE html>
<html lang="en">
<head>

    <style>
        body {
            font-family: Arial, sans-serif;
            max-width: 700px;
            margin: 50px auto;
            padding: 20px;
        }
        .student-card {
            background-color: #f0f0f0;
            padding: 15px;
            margin: 10px 0;
            border-radius: 5px;
            border-left: 4px solid #0066cc;
        }
        button {
            padding: 10px 20px;
            margin: 5px;
            font-size: 16px;
            background-color: #0066cc;
            color: white;
            border: none;
            border-radius: 5px;
            cursor: pointer;
        }
    </style>
</head>
<body>
    <h1>Student Management</h1>
    <button onclick="displayAll()">Display All Students</button>
    <button onclick="findStudent()">Find Student</button>
    <button onclick="showAverages()">Show Averages</button>
    
    <div id="output"></div>
    
    <script>
        // Array of student objects
        let students = [
            {
                name: "Sarah",
                age: 12,
                grade: 7,
                scores: [85, 90, 88],
                getAverage: function() {
                    const sum = this.scores.reduce((a, b) => a + b, 0);
                    return (sum / this.scores.length).toFixed(1);
                }
            },
            {
                name: "Alex",
                age: 13,
                grade: 8,
                scores: [92, 87, 95],
                getAverage: function() {
                    const sum = this.scores.reduce((a, b) => a + b, 0);
                    return (sum / this.scores.length).toFixed(1);
                }
            },
            {
                name: "Jordan",
                age: 12,
                grade: 7,
                scores: [78, 82, 80],
                getAverage: function() {
                    const sum = this.scores.reduce((a, b) => a + b, 0);
                    return (sum / this.scores.length).toFixed(1);
                }
            }
        ];
        
        function displayAll() {
            const output = document.getElementById('output');
            let html = '<h2>All Students</h2>';
            
            students.forEach(student => {
                html += `<div class="student-card">`;
                html += `<strong>${student.name}</strong> (Age ${student.age}, Grade ${student.grade})<br>`;
                html += `Scores: ${student.scores.join(', ')}<br>`;
                html += `Average: ${student.getAverage()}`;
                html += `</div>`;
            });
            
            output.innerHTML = html;
        }
        
        function findStudent() {
            const name = prompt("Enter student name:");
            const student = students.find(s => s.name.toLowerCase() === name.toLowerCase());
            
            const output = document.getElementById('output');
            if (student) {
                output.innerHTML = `
                    <h2>Student Found</h2>
                    <div class="student-card">
                        <strong>${student.name}</strong><br>
                        Age: ${student.age}<br>
                        Grade: ${student.grade}<br>
                        Average Score: ${student.getAverage()}
                    </div>
                `;
            } else {
                output.innerHTML = '<p>Student not found!</p>';
            }
        }
        
        function showAverages() {
            const output = document.getElementById('output');
            let html = '<h2>Student Averages</h2>';
            
            students.forEach(student => {
                html += `<p><strong>${student.name}:</strong> ${student.getAverage()}</p>`;
            });
            
            const totalAverage = students.reduce((sum, s) => sum + parseFloat(s.getAverage()), 0) / students.length;
            html += `<p><strong>Class Average:</strong> ${totalAverage.toFixed(1)}</p>`;
            
            output.innerHTML = html;
        }
    </script>
</body>
</html>

üí° Key Concepts

  • Object: A collection of properties
  • Property: A key-value pair in an object
  • Method: A function that belongs to an object
  • this: Refers to the current object
  • Dot notation: object.property
  • Bracket notation: object['property']
