<body>
    <h1>Array Manager</h1>
    <input type="text" id="itemInput" placeholder="Enter item">
    <button onclick="addItem()">Add Item</button>
    <button onclick="removeLast()">Remove Last</button>
    <button onclick="clearAll()">Clear All</button>
    <button onclick="sortItems()">Sort</button>
    
    <div id="list"></div>
    <div id="count"></div>
    
    <script>
        let items = ["apple", "banana", "orange"];
        
        function displayList() {
            const listDiv = document.getElementById('list');
            const countDiv = document.getElementById('count');
            
            if (items.length === 0) {
                listDiv.innerHTML = '<p>No items in list</p>';
            } else {
                listDiv.innerHTML = '';
                items.forEach((item, index) => {
                    const itemDiv = document.createElement('div');
                    itemDiv.className = 'list-item';
                    itemDiv.textContent = `${index + 1}. ${item}`;
                    listDiv.appendChild(itemDiv);
                });
            }
            
            countDiv.textContent = `Total items: ${items.length}`;
        }
        
        function addItem() {
            const input = document.getElementById('itemInput');
            const newItem = input.value.trim();
            
            if (newItem) {
                items.push(newItem);
                input.value = '';
                displayList();
            }
        }
        
        function removeLast() {
            if (items.length > 0) {
                items.pop();
                displayList();
            }
        }
        
        function clearAll() {
            items = [];
            displayList();
        }
        
        function sortItems() {
            items.sort();
            displayList();
        }
        
        // Display initial list
        displayList();
        
        // Allow Enter key to add item
        document.getElementById('itemInput').addEventListener('keypress', function(e) {
            if (e.key === 'Enter') {
                addItem();
            }
        });
    </script>
</body>
</html>

ğŸ’¡ Key Concepts

  • Array: A list of items
  • Index: Position in array (starts at 0)
  • push(): Adds item to end of array
  • pop(): Removes last item
  • forEach(): Loops through array
  • length: Number of items in array
  • sort(): Sorts array alphabetically
