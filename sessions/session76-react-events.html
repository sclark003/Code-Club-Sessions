Step 1: Handle input changes: 
function Input() {
  const [value, setValue] = useState('');
  
  const handleChange = (e) => {
    setValue(e.target.value);
  };
  
  return (
    <input 
      type="text" 
      value={value} 
      onChange={handleChange} 
    />
  );
}

Example: Here's a complete interactive form: 
// InteractiveForm.js
import { useState } from 'react';
import './InteractiveForm.css';

function InteractiveForm() {
  const [name, setName] = useState('');
  const [colour, setColour] = useState('#0066cc');
  const [message, setMessage] = useState('');
  
  const handleSubmit = (e) => {
    e.preventDefault();
    alert(`Hello ${name}! Your message: ${message}`);
  };
  
  return (
    <div className="form-container">
      <h2>Interactive Form</h2>
      <form onSubmit={handleSubmit}>
        <div className="form-group">
          <label>Name:</label>
          <input 
            type="text" 
            value={name}
            onChange={(e) => setName(e.target.value)}
            placeholder="Enter your name"
          />
        </div>
        
        <div className="form-group">
          <label>Favourite Colour:</label>
          <input 
            type="color" 
            value={colour}
            onChange={(e) => setColour(e.target.value)}
          />
          <div 
            className="colour-preview" 
            style={{ backgroundColor: colour }}
          >
            Preview
          </div>
        </div>
        
        <div className="form-group">
          <label>Message:</label>
          <textarea 
            value={message}
            onChange={(e) => setMessage(e.target.value)}
            placeholder="Enter a message"
            rows="4"
          />
        </div>
        
        <button type="submit">Submit</button>
      </form>
      
      {name && (
        <div className="preview">
          <h3>Preview:</h3>
          <p>Hello, <strong style={{ color: colour }}>{name}</strong>!</p>
          <p>Your message: {message}</p>
        </div>
      )}
    </div>
  );
}

export default InteractiveForm;
Create a component with:
  • A toggle button that shows/hides content
  • An input that changes background colour
  • A button that cycles through different messages

ğŸ’¡ Key Concepts

  • onClick: Handles click events
  • onChange: Handles input changes
  • onSubmit: Handles form submission
  • Event object: Contains information about the event
  • e.target.value: Gets the value from an input
  • e.preventDefault(): Stops default form behaviour
  • Controlled component: Input value controlled by state
