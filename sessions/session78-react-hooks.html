  • useEffect - For side effects (like API calls, timers)
  • useContext - For context

ğŸ’» Activity 2: useEffect Hook

Step 1: useEffect runs code after render: 
import { useState, useEffect } from 'react';

function Timer() {
  const [seconds, setSeconds] = useState(0);
  
  useEffect(() => {
    const interval = setInterval(() => {
      setSeconds(seconds => seconds + 1);
    }, 1000);
    
    return () => clearInterval(interval);
  }, []);
  
  return <div>Time: {seconds}s</div>;
}

ğŸ“‹ Complete Example Program

Example: Here's a complete example using hooks: 
// TimerAndCounter.js
import { useState, useEffect } from 'react';
import './TimerAndCounter.css';

function TimerAndCounter() {
  const [count, setCount] = useState(0);
  const [timer, setTimer] = useState(0);
  const [isRunning, setIsRunning] = useState(false);
  
  // Effect for timer
  useEffect(() => {
    let interval = null;
    if (isRunning) {
      interval = setInterval(() => {
        setTimer(timer => timer + 1);
      }, 1000);
    } else {
      clearInterval(interval);
    }
    return () => clearInterval(interval);
  }, [isRunning]);
  
  // Effect that runs when count changes
  useEffect(() => {
    document.title = `Count: ${count}`;
  }, [count]);
  
  const reset = () => {
    setTimer(0);
    setIsRunning(false);
  };
  
  return (
    <div className="timer-container">
      <h2>Timer and Counter</h2>
      
      <div className="section">
        <h3>Counter: {count}</h3>
        <button onClick={() => setCount(count + 1)}>Increment</button>
        <button onClick={() => setCount(count - 1)}>Decrement</button>
        <button onClick={() => setCount(0)}>Reset</button>
      </div>
      
      <div className="section">
        <h3>Timer: {timer}s</h3>
        <button onClick={() => setIsRunning(!isRunning)}>
          {isRunning ? 'Pause' : 'Start'}
        </button>
        <button onClick={reset}>Reset Timer</button>
      </div>
      
      <p className="info">Check the browser tab title - it shows the count!</p>
    </div>
  );
}

export default TimerAndCounter;

  • Shows how long the timer has been running

ğŸ’¡ Key Concepts

  • Hook: Special function starting with "use"
  • useEffect: Runs code after render
  • Dependency array: Controls when effect runs
  • Cleanup function: Code that runs when component unmounts
  • Side effect: Something that affects something outside the component
  • Empty array []: Effect runs only once on mount
