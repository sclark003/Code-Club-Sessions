üöÄ Activity 1: Creating Dictionaries

Step 1: A dictionary stores data as key-value pairs. Think of it like a real dictionary - you look up a word (key) to find its meaning (value): 
# Dictionary of student information
student = {
    "name": "Sarah",
    "age": 12,
    "grade": 7,
    "favourite_subject": "Maths"
}

print(student)
Step 2: Dictionaries use curly braces {} and colons : to separate keys and values
Dictionaries are like lists, but instead of using numbers (0, 1, 2...), you use keys (like "name", "age") to access values!

üíª Activity 2: Accessing Dictionary Values

Step 1: Access values using their keys: 
student = {
    "name": "Sarah",
    "age": 12,
    "grade": 7
}

print(student["name"])   # Sarah
print(student["age"])   # 12
print(student["grade"]) # 7
Step 2: You can also use .get() which is safer: 
print(student.get("name"))        # Sarah
print(student.get("height", "Unknown"))  # "Unknown" if key doesn't exist

üé® Activity 3: Modifying Dictionaries

Step 1: Add or change items: 
student = {"name": "Sarah", "age": 12}

# Add new item
student["grade"] = 7
print(student)

# Change existing item
student["age"] = 13
print(student)
Step 2: Remove items: 
student = {"name": "Sarah", "age": 12, "grade": 7}

# Remove an item
del student["grade"]
print(student)

# Or use pop()
age = student.pop("age")
print(f"Removed age: {age}")
print(student)

üìä Activity 4: Dictionary Methods

Step 1: Useful dictionary methods: 
student = {"name": "Sarah", "age": 12, "grade": 7}

# Get all keys
print(student.keys())

# Get all values
print(student.values())

# Get all items (key-value pairs)
print(student.items())

# Check if key exists
print("name" in student)  # True
print("height" in student)  # False

üéØ Activity 5: Looping Through Dictionaries

Step 1: Loop through keys, values, or both: 
student = {"name": "Sarah", "age": 12, "grade": 7}

# Loop through keys
for key in student:
    print(key)

# Loop through keys and values
for key, value in student.items():
    print(f"{key}: {value}")

# Loop through values only
for value in student.values():
    print(value)

üåü Activity 6: Practical Example

Step 1: Create a simple contact book: 
contacts = {}

while True:
    print("\nContact Book")
    print("1. Add contact")
    print("2. View contact")
    print("3. List all contacts")
    print("4. Quit")
    
    choice = input("Choose an option: ")
    
    if choice == "1":
        name = input("Name: ")
        phone = input("Phone: ")
        email = input("Email: ")
        contacts[name] = {"phone": phone, "email": email}
        print("Contact added!")
    
    elif choice == "2":
        name = input("Name to look up: ")
        if name in contacts:
            print(f"Phone: {contacts[name]['phone']}")
            print(f"Email: {contacts[name]['email']}")
        else:
            print("Contact not found!")
    
    elif choice == "3":
        for name, info in contacts.items():
            print(f"{name}: {info['phone']}, {info['email']}")
    
    elif choice == "4":
        break

Example: Here's a complete student management program: 
# student_manager.py
# A program to manage student information using dictionaries

# Initial student data
students = {
    "Sarah": {
        "age": 12,
        "grade": 7,
        "favourite_subject": "Maths",
        "email": "sarah@school.com"
    },
    "Alex": {
        "age": 13,
        "grade": 8,
        "favourite_subject": "Science",
        "email": "alex@school.com"
    },
    "Jordan": {
        "age": 12,
        "grade": 7,
        "favourite_subject": "English",
        "email": "jordan@school.com"
    }
}

print("Student Information Manager")
print("=" * 40)

while True:
    print("\nMenu:")
    print("1. View all students")
    print("2. Look up a student")
    print("3. Add a student")
    print("4. Update student information")
    print("5. Remove a student")
    print("6. Quit")
    
    choice = input("\nChoose an option (1-6): ")
    
    if choice == "1":
        print("\n" + "=" * 40)
        print("All Students")
        print("=" * 40)
        for name, info in students.items():
            print(f"\n{name}:")
            print(f"  Age: {info['age']}")
            print(f"  Grade: {info['grade']}")
            print(f"  Favourite Subject: {info['favourite_subject']}")
            print(f"  Email: {info['email']}")
    
    elif choice == "2":
        name = input("\nEnter student name: ")
        if name in students:
            info = students[name]
            print(f"\n{name}:")
            print(f"  Age: {info['age']}")
            print(f"  Grade: {info['grade']}")
            print(f"  Favourite Subject: {info['favourite_subject']}")
            print(f"  Email: {info['email']}")
        else:
            print(f"Student '{name}' not found!")
    
    elif choice == "3":
        name = input("Enter student name: ")
        if name in students:
            print(f"Student '{name}' already exists!")
        else:
            age = int(input("Enter age: "))
            grade = int(input("Enter grade: "))
            subject = input("Enter favourite subject: ")
            email = input("Enter email: ")
            students[name] = {
                "age": age,
                "grade": grade,
                "favourite_subject": subject,
                "email": email
            }
            print(f"Added {name} to the system!")
    
    elif choice == "4":
        name = input("Enter student name to update: ")
        if name in students:
            print("Enter new information (press Enter to keep current):")
            age = input(f"Age (current: {students[name]['age']}): ")
            if age:
                students[name]['age'] = int(age)
            
            grade = input(f"Grade (current: {students[name]['grade']}): ")
            if grade:
                students[name]['grade'] = int(grade)
            
            subject = input(f"Favourite Subject (current: {students[name]['favourite_subject']}): ")
            if subject:
                students[name]['favourite_subject'] = subject
            
            print(f"Updated {name}'s information!")
        else:
            print(f"Student '{name}' not found!")
    
    elif choice == "5":
        name = input("Enter student name to remove: ")
        if name in students:
            del students[name]
            print(f"Removed {name} from the system!")
        else:
            print(f"Student '{name}' not found!")
    
    elif choice == "6":
        print("\nFinal student list:")
        for name in students.keys():
            print(f"  - {name}")
        print("\nGoodbye!")
        break
    
    else:
        print("Invalid choice. Please try again.")

Create a program that:
  • Stores information about 3 students (name, age, favourite subject)
  • Allows you to look up a student by name
  • Allows you to update a student's information
  • Displays all students and their information

Example solution:

# student_info.py
students = {
    "Sarah": {"age": 12, "subject": "Maths"},
    "Alex": {"age": 13, "subject": "Science"},
    "Jordan": {"age": 12, "subject": "English"}
}

# Look up student
name = input("Enter student name: ")
if name in students:
    info = students[name]
    print(f"{name}: Age {info['age']}, Favourite: {info['subject']}")

# Update student
name = input("\nStudent to update: ")
if name in students:
    students[name]["age"] = int(input("New age: "))
    students[name]["subject"] = input("New subject: ")
    print("Updated!")

# Display all
print("\nAll students:")
for name, info in students.items():
    print(f"{name}: {info}")

üí° Key Concepts

  • Dictionary: A collection of key-value pairs stored in curly braces {}
  • Key: The identifier used to access a value (like "name")
  • Value: The data stored with a key (like "Sarah")
  • Key-Value Pair: A combination of a key and its value
  • .get(): Safely gets a value, with optional default if key doesn't exist
  • .keys(): Returns all keys in the dictionary
  • .values(): Returns all values in the dictionary
  • .items(): Returns all key-value pairs
