# utility_functions.py
import math

def rectangle_area(length, width):
    return length * width

def circle_area(radius):
    return math.pi * radius ** 2

def celsius_to_fahrenheit(celsius):
    return (celsius * 9/5) + 32

def is_prime(number):
    if number < 2:
        return False
    for i in range(2, int(math.sqrt(number)) + 1):
        if number % i == 0:
            return False
    return True

# Test the functions
print(f"Rectangle area (5x3): {rectangle_area(5, 3)}")
print(f"Circle area (radius 4): {circle_area(4):.2f}")

ğŸ’¡ Key Concepts

  • Function: A reusable block of code that performs a specific task
  • def: Keyword used to define a function
  • Parameter: A variable that receives a value when the function is called
  • Argument: The value passed to a function when it's called
  • return: Sends a value back from the function
  • Call: To execute a function by writing its name with parentheses
  • Default parameter: A parameter with a default value if none is provided
