üöÄ Activity 1: Creating a Component

Step 1: Create a new component file Welcome.js: 
function Welcome() {
  return (
    <div>
      <h2>Welcome to Code Club!</h2>
      <p>We're learning React!</p>
    </div>
  );
}

export default Welcome;
Step 2: Use it in App.js: 
import Welcome from './Welcome';

function App() {
  return (
    <div className="App">
      <Welcome />
    </div>
  );
}

üíª Activity 2: Using Props

Step 1: Pass data to components with props: 
// Welcome.js
function Welcome(props) {
  return (
    <div>
      <h2>Welcome, {props.name}!</h2>
      <p>You are {props.age} years old.</p>
    </div>
  );
}

// App.js
<Welcome name="Sarah" age={12} />

üìã Complete Example Program

Example: Here's a complete example with multiple components: 
// StudentCard.js
function StudentCard(props) {
  return (
    <div className="student-card">
      <h3>{props.name}</h3>
      <p>Age: {props.age}</p>
      <p>Grade: {props.grade}</p>
      <p>Favourite Subject: {props.subject}</p>
    </div>
  );
}

export default StudentCard;

// App.js
import './App.css';
import StudentCard from './StudentCard';

function App() {
  return (
    <div className="App">
      <h1>Student Directory</h1>
      <div className="students-container">
        <StudentCard 
          name="Sarah" 
          age={12} 
          grade={7} 
          subject="Maths" 
        />
        <StudentCard 
          name="Alex" 
          age={13} 
          grade={8} 
          subject="Science" 
        />
        <StudentCard 
          name="Jordan" 
          age={12} 
          grade={7} 
          subject="English" 
        />
      </div>
    </div>
  );
}

export default App;

// App.css
.students-container {
  display: flex;
  gap: 20px;
  flex-wrap: wrap;
  justify-content: center;
  margin-top: 20px;
}

.student-card {
  background-color: #f0f0f0;
  padding: 20px;
  border-radius: 10px;
  border: 2px solid #0066cc;
  min-width: 200px;
}

.student-card h3 {
  color: #0066cc;
  margin-top: 0;
}

üí° Key Concepts

  • Component: A reusable piece of UI
  • Props: Data passed to a component
  • JSX: HTML-like syntax in JavaScript
  • Import/Export: How to share components between files
  • Component name: Must start with capital letter
  • Props are read-only: Components can't change their props
