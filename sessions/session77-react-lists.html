Step 1: Use map() to render lists: 
function FruitList() {
  const fruits = ["apple", "banana", "orange"];
  
  return (
    <ul>
      {fruits.map((fruit) => (
        <li key={fruit}>{fruit}</li>
      ))}
    </ul>
  );
}
Always include a key prop when rendering lists - it helps React track which items have changed!

ğŸ“‹ Complete Example Program

Example: Here's a complete list component: 
// StudentList.js
import { useState } from 'react';
import './StudentList.css';

function StudentList() {
  const [students] = useState([
    { id: 1, name: "Sarah", age: 12, grade: 7, subject: "Maths" },
    { id: 2, name: "Alex", age: 13, grade: 8, subject: "Science" },
    { id: 3, name: "Jordan", age: 12, grade: 7, subject: "English" },
    { id: 4, name: "Taylor", age: 13, grade: 8, subject: "History" }
  ]);
  
  const [filterGrade, setFilterGrade] = useState('all');
  
  const filteredStudents = filterGrade === 'all' 
    ? students 
    : students.filter(s => s.grade === parseInt(filterGrade));
  
  return (
    <div className="student-list-container">
      <h2>Student Directory</h2>
      
      <div className="filter">
        <label>Filter by Grade:</label>
        <select 
          value={filterGrade} 
          onChange={(e) => setFilterGrade(e.target.value)}
        >
          <option value="all">All Grades</option>
          <option value="7">Grade 7</option>
          <option value="8">Grade 8</option>
        </select>
      </div>
      
      <div className="students-grid">
        {filteredStudents.map((student) => (
          <div key={student.id} className="student-card">
            <h3>{student.name}</h3>
            <p><strong>Age:</strong> {student.age}</p>
            <p><strong>Grade:</strong> {student.grade}</p>
            <p><strong>Subject:</strong> {student.subject}</p>
          </div>
        ))}
      </div>
      
      <p className="count">
        Showing {filteredStudents.length} of {students.length} students
      </p>
    </div>
  );
}

export default StudentList;

ğŸ’¡ Key Concepts

  • map(): Creates a new array by transforming each item
  • key: Unique identifier for list items
  • filter(): Creates a new array with filtered items
  • Conditional rendering: Showing different content based on conditions
  • Unique keys: Each item needs a unique key (usually an ID)
